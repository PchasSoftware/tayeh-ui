## Install and Usage

```bash
# installing package
npm install --save @tayeh/vue-components
```

#### in your Vue main.js file

```js
import Vue from 'vue';
import App from './App'
import TayehUi from '@tayeh/vue-components'

Vue.use(TayehUi)

new Vue({
  el: '#app',
  render: h => h(App)
})

```
